#define FIREBASE_HOST "smart-waste-management-afb88-default-rtdb.firebaseio.com"
#define FIREBASE_AUTH "AIzaSyAK3Ll6AL7IDQId-qzpp3mf37RZdmATdiQ"

FirebaseData firebaseData;

// ================= HARDWARE MAPPING =================
String slot1ID = "";
String slot2ID = "";
String slot3ID = "";

// ================= ULTRASONIC =================
#define TRIG1 5
#define ECHO1 18
#define TRIG2 4
#define ECHO2 19
#define TRIG3 23
#define ECHO3 22

// ================= LED PINS =================
#define RED1 2
#define GREEN1 27
#define RED2 13
#define GREEN2 14
#define RED3 21
#define GREEN3 25

#define BUZZER 15
#define BIN_HEIGHT 70.0

// ================= FUNCTIONS =================

void fetchMapping() {

  slot1ID = "";
  slot2ID = "";
  slot3ID = "";

  if (Firebase.getString(firebaseData, "/hardware_mapping/slot1"))
    slot1ID = firebaseData.stringData();

  if (Firebase.getString(firebaseData, "/hardware_mapping/slot2"))
    slot2ID = firebaseData.stringData();

  if (Firebase.getString(firebaseData, "/hardware_mapping/slot3"))
    slot3ID = firebaseData.stringData();

  Serial.println("---- Mapping ----");
  Serial.println("Slot1: " + slot1ID);
  Serial.println("Slot2: " + slot2ID);
  Serial.println("Slot3: " + slot3ID);
}

float getDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH, 30000);
  return duration * 0.034 / 2;
}

float calculateFill(float distance) {
  float fill = ((BIN_HEIGHT - distance) / BIN_HEIGHT) * 100;
  return constrain(fill, 0, 100);
}

String getStatus(float fill) {
  if (fill >= 80) return "critical";
  if (fill >= 70) return "needs_collection";
  return "normal";
}

bool updateLED(float fill, int redPin, int greenPin) {
  if (fill >= 80) {
    digitalWrite(redPin, HIGH);
    digitalWrite(greenPin, LOW);
    return true;
  } else {
    digitalWrite(redPin, LOW);
    digitalWrite(greenPin, HIGH);
    return false;
  }
}

// 🔥 SAFE UPDATE FUNCTION
void safeUpdateBin(String binID, float fill) {

  if (binID.length() < 10) return;

  // Check if binName exists (means created from dashboard)
  if (Firebase.getString(firebaseData, "/bins/" + binID + "/binName")) {

    Firebase.setFloat(firebaseData,
                      "/bins/" + binID + "/currentFillLevel",
                      fill);

    Firebase.setString(firebaseData,
                       "/bins/" + binID + "/status",
                       getStatus(fill));

    Serial.println("Updated: " + binID);

  } else {

    Serial.println("⚠ Bin not valid. Skipping update.");
  }
}

// ================= SETUP =================
void setup() {

  Serial.begin(115200);
  delay(2000);

  pinMode(TRIG1, OUTPUT); pinMode(ECHO1, INPUT);
  pinMode(TRIG2, OUTPUT); pinMode(ECHO2, INPUT);
  pinMode(TRIG3, OUTPUT); pinMode(ECHO3, INPUT);

  pinMode(RED1, OUTPUT); pinMode(GREEN1, OUTPUT);
  pinMode(RED2, OUTPUT); pinMode(GREEN2, OUTPUT);
  pinMode(RED3, OUTPUT); pinMode(GREEN3, OUTPUT);

  pinMode(BUZZER, OUTPUT);

  WiFi.begin(ssid, password);
  Serial.print("Connecting to WiFi");

  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }

  Serial.println("\nWiFi Connected!");

  Firebase.begin(FIREBASE_HOST, FIREBASE_AUTH);
  Firebase.reconnectWiFi(true);

  Serial.println("Firebase Ready");
}

// ================= LOOP =================
void loop() {

  fetchMapping();

  float fill1 = calculateFill(getDistance(TRIG1, ECHO1));
  float fill2 = calculateFill(getDistance(TRIG2, ECHO2));
  float fill3 = calculateFill(getDistance(TRIG3, ECHO3));

  bool alert = false;

  if (updateLED(fill1, RED1, GREEN1)) alert = true;
  if (updateLED(fill2, RED2, GREEN2)) alert = true;
  if (updateLED(fill3, RED3, GREEN3)) alert = true;

  digitalWrite(BUZZER, alert ? HIGH : LOW);

  // 🔥 SAFE UPDATE ONLY
  safeUpdateBin(slot1ID, fill1);
  safeUpdateBin(slot2ID, fill2);
  safeUpdateBin(slot3ID, fill3);

  Serial.println("Cycle Complete");
  delay(5000);
}
